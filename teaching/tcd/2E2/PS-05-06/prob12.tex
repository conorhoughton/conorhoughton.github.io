\noindent {\bf Useful facts:}\\
\begin{itemize}
\item Formula for exponentials with pure imaginary arguements
\begin{eqnarray}
e^{i\theta}&=&\cos{\theta}+i\sin{\theta}\cr
e^{-i\theta}&=&\cos{\theta}-i\sin{\theta}
\end{eqnarray}
\item Formula for sine and cos
\begin{eqnarray}
\cos{\theta}&=&\frac{e^{i\theta}+e^{-i\theta}}{2}\cr
\sin{\theta}&=&\frac{e^{i\theta}-e^{-i\theta}}{2i}
\end{eqnarray}
\end{itemize}
